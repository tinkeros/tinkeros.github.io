// RAX=RDX`RAX

  JNZ   @@05

  JMP   @@20
@@10:   MOV     RCX,RAX //RCX=pow

  MOV   U64 -16[RBP],RAX

  MOV   AX,U16 -24[RBP]

  FRNDINT

  F2XM1
  FLD1
  FADDP ST1,ST0

  FLD   U64 -8[RBP]
  FLD   U64 -16[RBP]

  TEST  RDX,RDX //was base neg?

  MOV   U64 -16[RBP],RCX
  FLD   U64 -16[RBP]
  FISTTP        U64 -16[RBP]
  TEST  U64 -16[RBP],1
  JZ    @@15    //was pow odd?
  FCHS
@@15:   FSTP    U64 -8[RBP]

@@20:   LEAVE
