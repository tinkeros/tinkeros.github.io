public U8 MOUSE_COM=0;
public I64 MOUSE_BAUD=1200;

public U0 ProbeSerialMouse()
{
  I64 baud,i,tmp_byte;
  U8 tmp_buf[17];
  MemSet(tmp_buf,0,17);
  for (i=1; i<=MAX_COMM_NUM; i++)
  {
    if (comm_ports[i].base)
    {
      "Checking COMM %d - ",i;
      CommInit7n1(i, 1200);
      CommPutChar(i,0xAA);
      Sleep(1);
      if (CommGetCharNoWait(i,tmp_byte))
      {
        if (*tmp_byte=='M')
        {
          "Got M 7n1 baud %d\n",baud;
          Sleep(100);
          if (CommGetCharNoWait(i,tmp_byte) && tmp_byte=='3')
          {
            "Found Serial Mouse on COMM %d\n",i;
            MOUSE_COM=i;
            goto probe_end;
          }
        }
      }
      " No device detected!\n";
    }
  }
  for (i=1; i<=MAX_COMM_NUM; i++)
  {
    if (comm_ports[i].base)
    {
      "Checking COMM %d - ",i;
      for (baud=1200; baud<=115200; baud*=2)
      {
        CommInit8n1(i, baud);
        CommPutChar(i,0xAA);
        Sleep(1);
        if (CommGetCharNoWait(i,tmp_byte))
        {
          if (*tmp_byte=='M')
          {
            "Got M 8n1 baud %d\n",baud;
            Sleep(100);
            if (CommGetCharNoWait(i,tmp_byte) && tmp_byte=='3')
            {
              "Found Serial Mouse on COMM %d\n",i;
              MOUSE_COM=i;
              MOUSE_BAUD=baud;
              goto probe_end;
            }
          }
        }
      }
      " No device detected!\n";
    }
  }
probe_end:
}


U0 Test7n1Mouse(I64 port=2, I64 msg_size=4)
{
  U8 tmp_byte, bytes[6];
  I64 idx=0;
  CommInit8n1(port, 115200);
  CommPutChar(port,0xAA);
  Sleep(1);
  // initial flush
  "Init port %d\n"
  ,port;
  while (CommGetCharNoWait(port,&tmp_byte))
  {
    //Nothing
  };
  "Flushed port %d\n"
  ,port;
  Sleep(1);
  while (TRUE)
  {
    Yield;
    if (CommGetCharNoWait(port,&bytes[idx++]))
      if (idx==msg_size)
      {
        "Got %d bytes!\n",msg_size;
        idx=0;
      }
  }
  "Done\n";
}
